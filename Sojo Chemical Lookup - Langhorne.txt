function filterProducts() {
  const q = document.getElementById('searchInput').value.toLowerCase();
  const cat = document.getElementById('categoryFilter').value;
  filtered = CHEMICALS.filter(c => {
    const matchQ = !q || c.name.toLowerCase().includes(q) || (c.company||'').toLowerCase().includes(q) || (c.code||'').toLowerCase().includes(q);

    return matchQ && matchCat;
  });
  activeIndex = -1;
  renderList();

function renderList() {
  const list = document.getElementById('productList');
  const badge = document.getElementById('countBadge');
  badge.textContent = `${filtered.length} product${filtered.length !== 1 ? 's' : ''} found`;
  list.innerHTML = '';
  if (!filtered.length) {
    list.innerHTML = '<div style="padding:16px;text-align:center;color:var(--text-dim);font-size:13px;">No products match your search.</div>';
    return;
  }
  filtered.forEach((c, i) => {
    const el = document.createElement('div');
    el.className = 'product-item' + (i === activeIndex ? ' active' : '');
    el.innerHTML = `<span class="product-item-name">${c.name}</span><span class="product-item-company">${c.company||''}</span><span class="product-item-use">${c.use||''}</span>`;
    el.onclick = () => selectProduct(i);
    list.appendChild(el);
  });
}

  activeIndex = i;

function updateDiamond(hfrp) {
  const parts = (hfrp || '').split('/');
  const h = parts[0] || '–';
  const f = parts[1] || '–';
  const r = parts[2] || '–';
  const s = parts[3] || '–';
  document.getElementById('svgHealth').textContent = h;
  document.getElementById('svgFire').textContent = f;
  document.getElementById('svgReact').textContent = r;
  document.getElementById('svgSpecial').textContent = s;
  document.getElementById('hfrpRaw').textContent = hfrp ? `H:${h}  F:${f}  R:${r}  S:${s}` : '';
}

// Init
filterProducts();
</script>
